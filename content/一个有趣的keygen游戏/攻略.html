    for (x = 0; x < name.length(); x++) {
        nameSum = ((nameSum + ((long) name.charAt(x))) * 3) - 64;
    }
    String sumString = Long.toString(nameSum);
    int finalSum = 0;
    for (x = 0; x < sumString.length(); x++) {
        finalSum += sumString.charAt(x) - 48;

    for (x = 0; x < serial.length(); x++) {

        return true;
    }
    return false;
}
}
